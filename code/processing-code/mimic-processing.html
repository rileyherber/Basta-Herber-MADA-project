This document covers all data loading, cleaning, and wrangling or the project examining the relationship between medication use and 30-day mortality in the MIMIC-IV database.

Primary outcome: 30-day mortality (binary: died within 30 days of admission)
Primary predictor: Medication use (drug name / medication class)
Covariates: Age at admission, gender

  1. Compute 30-Day Mortality
admissions_mortality <- admissions_clean |>
  left_join(
    patients_clean |> select(subject_id, dod, gender, anchor_age),
    by = "subject_id"
  ) |>
  mutate(
    days_to_death = as.numeric(difftime(dod, admittime, units = "days")),
    mortality_30day = case_when(
      !is.na(days_to_death) & days_to_death >= 0 & days_to_death <= 30 ~ 1L,
      TRUE ~ 0L
    )
  )

  count(mortality_30day) |>

  1. Assign Medication Classes
med_class_map <- tribble(
  ~pattern, ~med_class,
  "ASPIRIN","Antiplatelet",
  "HEPARIN","Anticoagulant",
  "INSULIN","Insulin",
  "FUROSEMIDE","Diuretic",
  "LISINOPRIL","ACE Inhibitor",
  "ATORVASTATIN","Statin"
)

assign_class <- function(drug_name){

  match <- med_class_map$med_class[
    str_detect(drug_name, med_class_map$pattern)
  ]

  if(length(match) > 0) match[1] else "Other"

}

prescriptions_clean <- prescriptions_clean |>
  mutate(med_class = map_chr(drug_std, assign_class))

  1. Build Admission Medication Flags
admission_med_flags <- prescriptions_clean |>
  filter(med_class != "Other") |>
  distinct(hadm_id, med_class) |>
  mutate(received = 1L) |>
  pivot_wider(
    names_from = med_class,
    values_from = received,
    values_fill = 0L
  ) |>
  clean_names()

  1. Final Analytic Dataset
analysis_df <- admissions_mortality |>
  inner_join(admission_med_flags, by = "hadm_id") |>
  mutate(
    age_at_admission = anchor_age,
    age_group = cut(
      age_at_admission,
      breaks = c(0,40,55,65,75,Inf),
      labels = c("<40","40–54","55–64","65–74","75+"),
      right = FALSE
    )
  ) |>
  select(
    subject_id,
    hadm_id,
    admittime,
    dischtime,
    los_days,
    gender,
    age_at_admission,
    age_group,
    race,
    mortality_30day,
    days_to_death,
    everything()
  )
