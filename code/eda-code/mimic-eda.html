4.4 Medication Class vs. 30-Day Mortality

This is the core exploratory analysis: comparing 30-day mortality rates across patients who did vs. did not receive each medication class.

med_mortality <- map_dfr(med_cols, function(col) {
  analysis_df |>
    group_by(received = .data[[col]]) |>
    summarise(n = n(), deaths = sum(mortality_30day), .groups = "drop") |>
    mutate(
      med_class  = str_to_title(str_replace_all(col, "_", " ")),
      mort_rate  = deaths / n,
      received   = if_else(received == 1L, "Received", "Not Received")
    )
})

med_mortality |>
  ggplot(aes(x = reorder(med_class, mort_rate), y = mort_rate, fill = received)) +
  geom_col(position = "dodge", width = 0.7) +
  scale_fill_manual(values = c("Received" = "#d62728", "Not Received" = "#aec7e8")) +
  scale_y_continuous(labels = scales::percent_format()) +
  coord_flip() +
  labs(title = "Unadjusted 30-Day Mortality Rate by Medication Class",
       subtitle = "Patients who received vs. did not receive each class",
       x = NULL, y = "30-Day Mortality Rate", fill = NULL) +
  theme(legend.position = "top")

Number of Medication Classes per Admission

Polypharmacy (receiving many medication classes) may itself be associated with disease severity and mortality.

analysis_df |>
  mutate(n_med_classes = rowSums(across(all_of(med_cols)))) |>
  group_by(n_med_classes) |>
  summarise(
    n         = n(),
    mort_rate = mean(mortality_30day)
  ) |>
  ggplot(aes(x = n_med_classes, y = mort_rate)) +
  geom_point(aes(size = n), color = "#d62728", alpha = 0.8) +
  geom_smooth(method = "loess", se = TRUE, color = "#1f77b4") +
  scale_y_continuous(labels = scales::percent_format()) +
  scale_size_continuous(name = "N Admissions") +
  labs(title = "Polypharmacy and 30-Day Mortality",
       x = "Number of Medication Classes Received",
       y = "30-Day Mortality Rate")
`geom_smooth()` using formula = 'y ~ x'

Mortality increased as the number of medications increased, this also could be explained by the increasing severity of illness as more medications were needed.
